### **모멘텀 알고리즘**
```typescript
function calculateMomentum(avgChange: number, positiveRatio: number) {
  if (avgChange > 5 && positiveRatio > 0.6) return 'bullish';   // 📈
  if (avgChange < -5 && positiveRatio < 0.4) return 'bearish';  // 📉
  return 'neutral';  // ⏸️
}
```
